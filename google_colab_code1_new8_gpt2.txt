def load_data():
    if not os.path.exists(CORPUS_PATH):
        print(f"Error: Folder not found: {CORPUS_PATH}")
        return

    files = [f for f in os.listdir(CORPUS_PATH) if f.endswith(".txt")]
    if not files:
        print("No .txt files found in corpus folder.")
        return

    for file_name in files:
        file_path = os.path.join(CORPUS_PATH, file_name)
        with open(file_path, "r", encoding="utf-8") as f:
            text = f.read().strip()
            if text:
                collection.upsert(documents=[text], ids=[file_name])

    print(f"Database ready. Total documents: {collection.count()}")

@app.get("/")

@app.post("/ask")
async def ask(request: Request):
    try:
        data = await request.json()
        user_query = (data.get("question") or "").strip()

        if not user_query:
            return JSONResponse(status_code=400, content={"error": "Question is empty"})

        results = collection.query(query_texts=[user_query], n_results=3)
        docs = results.get("documents", [[]])

        if not docs or not docs[0]:
            return JSONResponse(content={"answer": "No relevant corpus documents were found."})

        context = "\n\n".join(docs[0])

        prompt = f"""Answer the user's question only using the context below.
If the answer is not in the context, say that the information is not available in the corpus.

Question:
{user_query}

Context:
{context}
"""

        response = client_genai.models.generate_content(
            model="gemini-2.5-flash",
            contents=prompt
        )

        answer_text = getattr(response, "text", None)

        if not answer_text:
            return JSONResponse(
                status_code=500,
                content={"error": "Gemini returned no text output", "debug": str(response)}
            )

        return JSONResponse(content={"answer": answer_text})

    except Exception as e:
        return JSONResponse(status_code=500, content={"error": f"Backend processing error: {str(e)}"})
